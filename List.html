def [] (index)
        if index.is_a?Integer
                if index == 0
                        return @head.value
                elsif index == (@size - 1)
                        return @tail.value
                elsif index < (@size - 1) && index > 0
                             iterator = @head
                        index.times(iterator = iterator.next)
                        return iterator.value
                else
                        return nil
                end
        end
end

        return nil unless @size > 0
        aux = @head
        until aux.nil?
                yield aux.value
                aux = aux.next
        end

def extract_head

        if (@size == 0)

def insert_head(value)
        
        nodo = Node.new(value, nil, nil)
        
        if (@size == 0)
                @tail = nodo
                nodo.next = nil
        else
                @head.prev = nodo
                nodo.next = @head
        end

        @head = nodo
        nodo.prev = nil
        @size = @size + 1

end
