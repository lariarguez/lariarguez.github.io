def emisionesGases
 
        nodo = @listaAlimentos.head
        suma_gei = 0

        while(nodo != nil)
                suma_gei += nodo.value.GasesEfectoInvernadero
                nodo = nodo.next
        end

        return (suma_gei).round(2)
end

def to_s
        
        suma = 0
        @listaAlimentos.get_size.times do |i|
                suma += @listaAlimentos[i].ValorEnergetico
        end
        
        return (suma).round(2)

end
